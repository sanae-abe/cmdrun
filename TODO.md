use strsim::levenshtein;

pub struct TypoDetector {
    commands: Vec<String>,
    threshold: usize, // Default: 2
}

impl TypoDetector {
    pub fn suggest(&self, input: &str) -> Option<Vec<String>> {
        let mut suggestions: Vec<(String, usize)> = self.commands
            .iter()
            .map(|cmd| (cmd.clone(), levenshtein(input, cmd)))
            .filter(|(_, distance)| *distance <= self.threshold)
            .collect();

        suggestions.sort_by_key(|(_, distance)| *distance);

        if suggestions.is_empty() {
            None
        } else {
            Some(suggestions.iter().map(|(cmd, _)| cmd.clone()).collect())
        }
    }
}
